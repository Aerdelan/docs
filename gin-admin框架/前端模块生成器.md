```go
package main

import (
	"bytes"
	"flag"
	"fmt"
	"io"
	"os"
	"path"
	"strings"

	"github.com/jinzhu/inflection"
)

// 定义全局变量，用于存储命令行参数
var m string
var s string
var c string

// m=[模块名(大驼峰)] s=[结构体名(大驼峰)] c=[结构注释]
func main() {
	// 解析命令行参数
	flag.StringVar(&m, "m", m, "模块名(大驼峰)")
	flag.StringVar(&s, "s", s, "结构体名(大驼峰)")
	flag.StringVar(&c, "c", c, "注释")
	flag.Parse()

	// 检查参数是否为空
	if m == "" {
		fmt.Println("-m 模块名不能为空")
		return
	}
	if s == "" {
		fmt.Println("-s 结构体名不能为空")
		return
	}
	if c == "" {
		fmt.Println("-c 注释不能为空")
		return
	}

	// 如果目标目录存在则提示后退出
	if FileIsExisted(path.Join("src", "pages", ConvertUpperCamelToLowerCamel(m), ConvertUpperCamelToLowerCamel(s))) {
		fmt.Println("目标模块已存在")
		return
	}

	// 如果目标接口文件存在则提示后退出
	if FileIsExisted(path.Join("src", "services", ConvertUpperCamelToLowerCamel(m), fmt.Sprintf("%s.ts", ConvertUpperCamelToLowerCamel(s)))) {
		fmt.Println("目标接口文件已存在")
		return
	}

	// 创建目标目录
	_ = MakeDir(path.Join("src", "pages", ConvertUpperCamelToLowerCamel(m), ConvertUpperCamelToLowerCamel(s)))

	// 复制模板目录到目标目录
	err := CopyDir(path.Join("src", "pages", "tpl", "struct"), path.Join("src", "pages", ConvertUpperCamelToLowerCamel(m), ConvertUpperCamelToLowerCamel(s)))
	if err != nil {
		fmt.Println(err)
		return
	}

	// 检查是否存在 services 目录，如果不存在则创建
	servicesDir := path.Join("src", "services", ConvertUpperCamelToLowerCamel(m))
	if _, err := os.Stat(servicesDir); os.IsNotExist(err) {
		err = os.MkdirAll(servicesDir, os.ModePerm)
		if err != nil {
			fmt.Println(err)
			return
		}
	}

	// 复制 services 文件并重命名
	srcFile := path.Join("src", "pages", "tpl", "services.ts")
	dstFile := path.Join(servicesDir, fmt.Sprintf("%s.ts", ConvertUpperCamelToLowerCamel(s)))
	err = copyFile(srcFile, dstFile)
	if err != nil {
		fmt.Println(err)
		return
	}

	// 替换文件夹下的文本内容
	fs, err := WalkDir(path.Join("src", "pages", ConvertUpperCamelToLowerCamel(m), ConvertUpperCamelToLowerCamel(s)), "")
	if err != nil {
		return
	}
	for _, filePath := range fs {
		err := replacePlaceholders(filePath, m, s, c)
		if err != nil {
			fmt.Println(err)
			return
		}
	}

	serviceFile := path.Join(servicesDir, fmt.Sprintf("%s.ts", ConvertUpperCamelToLowerCamel(s)))
	err = replacePlaceholders(serviceFile, m, s, c)
	if err != nil {
		fmt.Println(err)
		return
	}

	fmt.Println("生成成功")
}

// 复制文件
func copyFile(src, dst string) error {
	sourceFile, err := os.Open(src)
	if err != nil {
		return err
	}
	defer sourceFile.Close()

	destinationFile, err := os.Create(dst)
	if err != nil {
		return err
	}
	defer destinationFile.Close()

	_, err = io.Copy(destinationFile, sourceFile)
	return err
}

// 将字符串转换为小写复数形式
func toLowerPlural(v, sep string) string {
	ss := strings.Split(ToLowerUnderlinedNamer(v), "_")
	if len(ss) > 0 {
		ss[len(ss)-1] = ToPlural(ss[len(ss)-1])
	}
	return strings.Join(ss, sep)
}

// 将字符串转换为复数形式
func ToPlural(v string) string {
	return inflection.Plural(v)
}

// 将字符串转换为小写下划线命名形式
func ToLowerUnderlinedNamer(name string) string {
	const (
		lower = false
		upper = true
	)

	if name == "" {
		return ""
	}

	// 定义常见缩略词
	var commonInitialismsReplacer *strings.Replacer
	var commonInitialismsForReplacer []string
	var commonInitialisms = []string{"API", "ASCII", "CPU", "CSS", "DNS", "EOF", "GUID", "HTML", "HTTP", "HTTPS", "ID", "IP", "JSON", "LHS", "QPS", "RAM", "RHS", "RPC", "SLA", "SMTP", "SSH", "TLS", "TTL", "UID", "UI", "UUID", "URI", "URL", "UTF8", "VM", "XML", "XSRF", "XSS"}
	for _, initialism := range commonInitialisms {
		commonInitialismsForReplacer = append(commonInitialismsForReplacer, initialism, strings.Title(strings.ToLower(initialism)))
	}
	commonInitialismsReplacer = strings.NewReplacer(commonInitialismsForReplacer...)

	var (
		value                                    = commonInitialismsReplacer.Replace(name)
		buf                                      = bytes.NewBufferString("")
		lastCase, currCase, nextCase, nextNumber bool
	)

	// 遍历字符串，插入下划线
	for i, v := range value[:len(value)-1] {
		nextCase = bool(value[i+1] >= 'A' && value[i+1] <= 'Z')
		nextNumber = bool(value[i+1] >= '0' && value[i+1] <= '9')

		if i > 0 {
			if currCase == upper {
				if lastCase == upper && (nextCase == upper || nextNumber == upper) {
					buf.WriteRune(v)
				} else {
					if value[i-1] != '_' && value[i+1] != '_' {
						buf.WriteRune('_')
					}
					buf.WriteRune(v)
				}
			} else {
				buf.WriteRune(v)
				if i == len(value)-2 && (nextCase == upper && nextNumber == lower) {
					buf.WriteRune('_')
				}
			}
		} else {
			currCase = upper
			buf.WriteRune(v)
		}
		lastCase = currCase
		currCase = nextCase
	}

	buf.WriteByte(value[len(value)-1])

	s := strings.ToLower(buf.String())
	return s
}

// 将字符串转换为小写驼峰命名形式
func ConvertUpperCamelToLowerCamel(s string) string {
	if len(s) == 0 {
		return s
	}
	return strings.ToLower(s[:1]) + s[1:]
}
```
